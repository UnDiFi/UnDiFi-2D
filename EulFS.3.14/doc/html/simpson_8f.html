subroutine SIMPSON ( integer  NDIM,
integer  NOFVERT,
integer  NDOF,
double precision, dimension(ndof,nofvert)  ZROE,
double precision, dimension(ndim,nofvert)  XYZDOT,
double precision, dimension(ndim)  FACN,
double precision, dimension(ndof,nofvert)  FLXN,
external  FLUXF 
)
Purpose

Compute the integral of the flux FLUXF through a linear/triangular face using Simpson's rule

Parameters
[in]NDIMdimension of the space
[in]NOFVERTnof vertices for a cell = NDIM+1
[in]NDOFnof degrees of freedom in vector ZROE, also leading dimension of FLXN
[in]ZROERoe's parameter vector
[in]XYZDOTthe cartesian components of the grid velocity
[in]FACNthe cartesian components of the scaled inward normal to the face
[out]FLXNthe eulerian (inviscid) nodal fluxes: the first NDIM columns return the fluxes in the NDIM vertices of the face, the last one (NOFVERT) is used to store the face value computed according to Simpson's rule
[in]FLUXFis the subroutine used to compute the flux the calling sequence is: 
 FLUXF((NDIM,FACN,ZROE(1,I),XYZDOT,FLXN(1,I))
